\subsection{Operadores Lógico, Relacional e Condicional}

\begin{frame}
    \frametitle{Operadores Lógico, Relacional e Condicional}
    \begin{itemize}
            \item <1-> Utilizados em expressões onde o resultado retornado ou a característica é booleana.
            \item <2-> O operador de negação ! retorna o complemento (contrário) de um valor booleano.
    \end{itemize}
    \begin{block}{Operadores Lógico, Relacional e Condicional}
        Os operadores relacionais ==, !=, $<$, $>$, $<$=, $>$=, resultam em um valor
        booleano e representam igual, não igual ou diferente, menor, maior,
        menor ou igual e maior ou igual, respectivamente.
    \end{block}
\end{frame}

\subsection{Operação de Atribuição}

\begin{frame}
    \frametitle{Operação de Atribuição}
    \begin{itemize}
            \item <1-> Divididos entre simples e compostos;
            \item <2-> Utilizados na designação de um valor para uma variável;
            \item <3-> O operador = representa a atribuição simples;
    \end{itemize}
    \begin{block}{Operadores de atribuição compostos}
        Os operadores +=, -=, *=, /=, \%=, \&=, |=, \^=, $<<$= e $>>$=
        representam a atribuição composta, que normalmente atuam como um
        atalho na construção de uma expressão.
    \end{block}
\end{frame}
