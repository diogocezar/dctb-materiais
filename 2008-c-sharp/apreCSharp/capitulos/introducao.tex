\section[Introdução]{Introdução}
\begin{frame}
    \frametitle{Introdução}
    \begin{itemize}
        \item <1-> O que é uma plataforma de desenvolvimento?
        \item <2-> Platadorma .NET;
        \item <3-> Características:
            \begin{itemize}
                \item <4-> Independência de linguagem de programação (C\#, J\#, C++, VB);
                \item <5-> Reutilização de código (DLL e Outra Bibliotecas);
                \item <6-> Tempo de execução compartilhado (\emph{runtime} compartilhado entre as linguagens);
                \item <7-> Sistemas auto-explicativos e controle de versões;
                \item <8-> Simplicidade na resolução de problemas complexos;
                \item <9-> Multi-plataforma (independente do sistema operacional): CLR;
            \end{itemize}
    \end{itemize}
\end{frame}

\subsection[Arquitetura .NET]{Arquitetura .NET}
\begin{frame}
    \frametitle{Arquitetura .NET}
        \begin{block}<1->{CLR (\emph{Commom Language Runtime})}
            É o ambiente de execução das aplicações .NET
        \end{block}
        \begin{block}<2->{CLS (\emph{Common Language Specification})}
            É um conjunto de regras que têm como objetivo gerar uma
            só resultante para a compilação de qualquer uma das
            linguagens suportadas pela plataforma .NET
        \end{block}
        \begin{block}<3->{BCL (\emph{Base Classe Library})}
            É uma biblioteca de componentes de software reutilizáveis
        \end{block}
\end{frame}
