\section{Apresentação do Projeto Integrador}

\subsection{Etapas para apresentação do projeto integrador}

\begin{frame}
    \frametitle{Etapas para apresentação do projeto integrador}
    \begin{itemize}
        \item Introdução;
        \item Metodologia;
        \item Tecnologia utilizada;
        \item O Sistema;
        \item Considerações finais.
    \end{itemize}
\end{frame}

\subsubsection{Introdução}
\begin{frame}
    \frametitle{Introdução}
    \begin{itemize}
        \item O que é?
        \begin{itemize}
            \item Pequena descrição em tópicos do sistema.
        \end{itemize}
        \item Quem auxiliará?
        \begin{itemize}
            \item Pequena descrição em tópicos de quem o sistema
            auxiliará (atores).
        \end{itemize}
        \item A empresa?
        \begin{itemize}
            \item Descrição da empresa para qual será desenvolvido o
            sistema.
        \end{itemize}
        \item Por que informatizar?
        \begin{itemize}
            \item Justificativa da necessidade do sistema
            desenvolvido no âmbito empresarial.
        \end{itemize}
    \end{itemize}
\end{frame}

\subsubsection{Metodologia}
\begin{frame}
    \frametitle{Metodologia}
    \begin{itemize}
        \item Qual a metodologia de desenvolvimento utilizada no processo?
        \begin{itemize}
            \item Pequena descrição em tópicos da metodologia.
        \end{itemize}
        \item Artefatos gerados:
        \begin{itemize}
            \item Descrição dos principais artefatos gerados.
        \end{itemize}
    \end{itemize}
\end{frame}

\subsubsection{Tecnologia utilizada}
\begin{frame}
    \frametitle{Tecnologia utilizada}
    \begin{itemize}
        \item Linguagem de programação:
        \begin{itemize}
            \item Descrever as funcionalidades aplicadas ao sistema disponíveis pela linguagem de
            programação.
        \end{itemize}
        \item Persistência de dados:
        \begin{itemize}
            \item Descrever as funcionalidades do SGBD utilizado
            para desenvolvimento do projeto.
        \end{itemize}
    \end{itemize}
\end{frame}

\subsubsection{O sistema}
\begin{frame}
    \frametitle{O sistema}
    \begin{itemize}
        \item Objetivos do sistema;
        \item Levantamento de requisitos:
        \begin{itemize}
            \item Qual a base para o desenvolvimento do projeto?
        \end{itemize}
        \item Pontos fortes:
        \begin{itemize}
            \item Descrever os pontos fortes do sistema desenvolvido
            (qual a contribuição para a empresa?).
        \end{itemize}
        \item Pontos fracos:
        \begin{itemize}
            \item Descrever os pontos fracos do sistema desenvolvido
            (que mudanças ou complicações o sistema pode gerar?).
        \end{itemize}
        \item Facilidades encontradas no desenvolvimento do sistema;
        \item Dificuldades encontradas no desenvolvimento do
        sistema;
        \item Protótipos de tela gerados.
    \end{itemize}
\end{frame}
