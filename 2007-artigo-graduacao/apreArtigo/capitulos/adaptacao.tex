\section{Camada de Adaptação}

\begin{frame}
    \frametitle{Cálculo da Relevância de uma Página}

    \begin{equation}
        \omega(o, d, g) = \frac{F_{odg}}{\sum_{t=1}^{n}F_{otg}}
        \label{eq:relevancia}
    \end{equation}

    \emph{onde:}

    \begin{itemize}
        \item $\omega$ é a relevância daquela página em relação às outras;
        \item $n$ é o número de páginas destino a partir daquela
        origem.
    \end{itemize}
\end{frame}
