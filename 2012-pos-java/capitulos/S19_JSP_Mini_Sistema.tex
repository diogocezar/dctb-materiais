\section[JSP Mini Sistema de Exemplo]{JSP Mini Sistema de Exemplo}

\begin{frame}[c,allowframebreaks,fragile]
    \frametitle{JSP Mini Sistema de Exemplo}
    \begin{itemize}
      \item Sistema de gerenciamento de alunos (CRUD);
      \item Um aluno pode ter mais de um endereço;
      \item Este mini sistema de exemplo extende as tecnologias aprendidas neste curso. Suas principais características são:
      \begin{itemize}
       \item \textbf{Hibernate} - Camada de persistência de dados;
       \item \textbf{Ajax}  - Exclusão de um Registro;
       \item \textbf{Sessão} - Sistema de Login;
       \item \textbf{XHTML} - Construção do Layout;
       \item \textbf{CSS} - Construção do Layout;
       \item \textbf{Tableless} - Construção do Layout;
       \item \textbf{JSTL} - Exibições e Cadastros;
       \item \textbf{Conceitos de MVC} - Modelo e visão pelo Hibernate e \textit{AlunoController} como controlador;
       \item \textbf{JavaScript e jQuery} - Método de inclusão de endereços e Validações;
       \item \textbf{Master Page} - Todas as páginas possuem o mesmo layout;
      \end{itemize}
    \end{itemize}
\end{frame}
