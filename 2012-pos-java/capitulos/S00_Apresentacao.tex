\section[Apresentacao]{Apresentação}

\begin{frame}
    \frametitle{Objetivo}
    \begin{itemize}
      \item proporcionar o aprendizado completo e abrangente das técnicas
e conceitos envolvidos no desenvolvimento de sistemas computacionais para
Internet, através de sua implementação.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Ementa}
    \begin{itemize}
      \item introdução à API Servlets, Ciclo de vida de servlets, API JSP,
Estrutura de páginas JSP, Ciclo de vida de JSPs, Controle de Headers HTTP,
Cookies, Controle de sessões, Integração com JavaBeans, Controle de Acesso,
JSTL, Expression Language, Deployment de uma aplicação web J2EE.
    \end{itemize}
\end{frame}

       \item Padrões para entrega (documento separado);
       \item Data limite: 01/12/2012;
       \item Valor: 40 pontos;
      \end{itemize}
      \item Projeto final;
      \begin{itemize}
	\item Projeto envolvendo todas as tecnologias apresentadas;

	\item Data de entrega: 01/01/2013;
	\item Valor: 60 pontos;
