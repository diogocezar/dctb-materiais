	\begin{itemize}
	 \item hibernate gera o banco na primeira execussão;
	 \item criação do banco com SQL em anexo no projeto;
	 \item recuperação do banco por seu \textit{backup};
	\end{itemize}
	\item utilização de AJAX em alguma parte do sistema;
	\item utilização de Sessão e Cookie para sistema de login;
	\item utilização de JSTL;
	\item utilização de Beans;
	\item organização do projeto: MVC;
