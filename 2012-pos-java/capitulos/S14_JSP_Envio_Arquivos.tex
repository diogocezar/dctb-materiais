\section[JSP Envio de Arquivos]{JSP Envio de Arquivos}

\begin{frame}[c,allowframebreaks,fragile]
    \frametitle{JSP Envio de Arquivos}
    \begin{itemize}
      \item \textit{UploadBean} é um componente JAVA (JavaBean) que permite
upload de arquivos;
      \item este bean pode ser integrado em qualquer aplicação JSP/Servlet;
      \item pode-se armazenar arquivos de upload em uma Pasta, arquivo ZIP, um
Banco de Dados ou em Memória;

      \framebreak

      \item restrições como limite de tamanho de arquivo, listas negras/brandas
e número máximo de arquivos de upload estão disponíveis;
      \item \textit{UploadBean} é gratuito para uso não-comercial;
      \item
\url{http://www.javazoom.net/jzservlets/uploadbean/uploadbean_pt.html}

      \framebreak
