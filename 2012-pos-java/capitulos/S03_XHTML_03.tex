    \frametitle{XHTML - Manipulando Tabelas}
    \begin{itemize}
      \item os elementos \texttt{<table>} e \texttt{</table>} são responsáveis
pela manutenção das tabelas nesta linguagem;
      \item uma célula pode se estender por mais de uma coluna, para isto deve
ser utilizado o atributo ``colspan'' seguido do número de colunas que a célula
deve se expandir;
      \item caso a célula tenha de ocupar mais de uma linha o atributo a ser
utilizado é o ``rowspan'', ambos os atributos devem ser inseridos dentro de uma
declaração \texttt{<td>} e \texttt{</td>}.
      \item na construção de uma tabela podemos utilizar alguns elementos;
      \item uma tabela ainda possui alguns atributos importantes;
    \end{itemize}
\end{frame}

\begin{frame}[c,allowframebreaks]
    \frametitle{XHTML - Imagens}
    \begin{itemize}
      \item atualmente os navegadores aceitam três formatos de figuras: GIF,
JPEG e PNG;
      \item utiliza-se o elemento \texttt{<img>};
      \item atributo ``alt'' utilizado para fornecer um texto alternativo
quando a imagem não puder ser visualizada;
      \item atributo ``src'' utilizado para indicar a localização da
figura;
      \item imagens png e gif podem ter o fundo transparente;
      \item as imagens referêntes ao layout de uma página não são manipuladas
pela tag \texttt{<img>}, são adicionadas no arquivos CSS;
      \item As imagens possuem alguns atributos importatens;
    \end{itemize}
\end{frame}

\begin{frame}[c,allowframebreaks]
    \frametitle{XHTML - Formulários}
    \begin{itemize}
      \item os formulários são utilizados para o envio de informações ao
servidor ou para a interação da página com o usuário através do uso de
JavaScripts;
      \item os dados podem ser submetidos através de dois métodos \textbf{GET} e
\textbf{POST}:
      \begin{itemize}
       \item \textit{GET}: os dados serão anexados à URL e enviados ao servidor;
       \item \textit{POST}: envia os dados junto com as demais informações
contidas no cabeçalho do HTTP;
      \end{itemize}
      \framebreak
      \item ao contrário do \textit{POST} que pode manipula qualquer quantidade
de informações, o \textit{GET} fica limitado pelo tamanho máximo permitido para
uma URL;
      \item O \textit{GET} é menos seguro, mas seu uso permite o reenvio de um
formulário sem que o mesmo seja re-digitado;
      \item O \textit{GET} pode funcionar como um controle para exibição do
conteúdo de uma mesma página;
      \item para a construção dos formulários devemos fazer uso dos elementos
\texttt{<form>} e \texttt{</form>}, que possuem os seguintes atributos:
    \end{itemize}
\end{frame}
