\begin{frame}[c,allowframebreaks]
    \frametitle{XHTML - Introdução}
    \begin{itemize}
      \item XHTML (\textit{EXtensible HyperText Markup Language});
      \item uma linguagem quase igual ao HTML original, mas que é capaz de
apresentar a "flexibilidade" da linguagem XML;
      \item é uma aplicação da XML criada para substituir a linguagem HTML;
      \item as marcações (tags) e atributos foram criadas com base nas do HTML
4.01 e suas regras;
      \item XHTML 1.0 é uma recomendação da W3C e sua versão atual data de 26 de
janeiro de 2000.
      \framebreak
      \item Vantagens:
      \begin{itemize}
       \item compatibilidade com as futuras aplicações;
       \item código consistente;
       \item facilidade de edição;
       \item carregamento rápido.
      \end{itemize}
      \framebreak
      \item Regras básicas do XHTML 1.0:
      \begin{itemize}
       \item fechar todas as tags;
       \item todas as tags devem ser escritas com letras minúsculas;
       \item todos os atributos devem conter aspas;
       \item morre o atributo name (substituido pelo id);
       \item atributos sem valor foram descontinuados, utiliza-se um novo
padrão;
       \item todos os atributos HTML devem estar aninhados;
       \framebreak
       \lstinputlisting[language=HTML,label=html-basico,
caption={Novidades XHTML}]{codes/NovidadesXHTML.txt}
      \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{XHTML - Conceitos Básicos - Outros}
    \begin{itemize}
     \item além dos elementos estudados, podemos utilizar outros dois elementos
de estilo para formatar nosso documento XHTML:
      \begin{itemize}
	\item o primeiro é o \texttt{<br />}: utilizado para inserir uma quebra
de linha;
	\item o segundo é o \texttt{<hr />}: utilizado para inserir uma linha
horizontal na apresentação do documento.
      \end{itemize}
    \end{itemize}
