\begin{frame}[c,allowframebreaks]
    \frametitle{Histórico}
    \begin{itemize}
      \item 1995 - Netscape:
      \begin{itemize}
	\item lança JavaScript;
	\item métricas (especificação) a serem seguidas pela linguagem;
	\item presente até os dias atuais;
	\item permite a interação com os objetos HTML no lado cliente;
	\item pode alterar características dos objetos que compões o documento
(textos, formulários, imagens) em tempo de execução;
	\item lança junto com o JavaScript o DOM(Document Object Model):
mapeia os elementos HTML em formato de árvore.
	\item mapeia os elementos HTML em formato de árvore.
	\framebreak
	\item o JavaScript também foi implementado por outras empresas:
	\begin{itemize}
	 \item Microsoft lança o JScript;
	 \item Adobe lança o ActionScript.
	\end{itemize}
      \end{itemize}
      \framebreak
      \item em 1998, Jhonatan colaborador da Microsoft, implementou a função que
daria vida ao Ajax: XMLHTTP;
      \begin{itemize}
       \item apesar do nome, não tem muito a ver com XML;
       \item XMLHTTP: Comunicação com o servidor de forma assíncrona;
       \item na época não teve destaque: não tinha ficado claro seu exato
funcionamento.
      \framebreak
      \end{itemize}

implementou a função XMLHttpRequest;
      \begin{itemize}
       \item ``Mesma'' coisa com nome diferente!
      \end{itemize}
      \item somente em 2004, o mundo se rendeu a essa nova funcionalidade:
      \begin{itemize}
       \item mágicas feitas pelo Google: Gmail!
       \item nasce nesse momento o conceito de programação Ajax
(Asynchronous Javascript And XML);
       \item não era um conceito novo, mas foi com o Google que teve destaque.
      \end{itemize}
      \framebreak
      \item em 2005, Google lança o Google Maps:
      \begin{itemize}
       \item foi a primeira aplicação web com características de uma aplicação
Desktop;
      \end{itemize}
      \end{itemize}
\end{frame}

\section[HTML5]{HTML5}

\begin{frame}[c,allowframebreaks]
    \frametitle{Surgimento do HTML5}
    \begin{itemize}
    \item em 2004, representantes das empresas: Mozzila, Apple e Opera se reunem
para criar um grupo paralelo a W3C;
    \item \textit{The Web Hypertext Application Technology Working Group}
(WHAT-WG);
    \item começam a desenvolver uma codificação diferente a proposta pela W3C
com o XHTML 2.0;
    \item esse grupo tomou como base a evolução e não a revolução;
    
    \framebreak
    
    \item objetivo: garantir compatibilidade do mesmo código para navegadores
diferentes;
    \item proposta do HTML5: especificar detalhadamente como interpretar
possíveis erros para que seja possível manter a maior compatibilidade possível;

    \framebreak
    
    \item HTML5 já pode ser usado?
    
    \begin{itemize}
     \item navegadores modernos já implementam boa parte de sua especificação;
     \item Mozzila, Chrome, Opera e Safari;
     \item Internet Explorer NÃO! Alias, não implementa 100\% nem das diretrizes
da W3C propostas em 1999;
    \end{itemize}
    
    \item o que vai mudar?
    
    \begin{itemize}
      \item organização e simplificação de código;
      \item acesso nativo a audio e vídeo;
      \item possibilidade de interação CANVAS (desenho) no próprio documento
HTML;
      \item possibilidade de estilos diferenciados e modernos (CSS3);
    \end{itemize}
    
    \framebreak

    \item então vamos aprender HTML5?
    
    \begin{itemize}
      \item não (só ele)! O mercado ainda não esta preparado para migrar;
      \item códigos HTML5 não são interpretados por todos navegadores;
      \item nem todos os recursos estão implementados;
    \end{itemize}
    
    \item Quando usar HTML5?

    \begin{itemize}
      \item Sistemas específicos para rodar em navegadores específicos;
    \end{itemize}   
    \end{itemize}
\end{frame}
