\begin{frame}
    \frametitle{Considerações}
    \begin{itemize}
		\item experimentos direcionados para a validação do modelo: principais funcionalidades;
		\item experimento 1:
		\begin{itemize}
			\item validação da paralelização direta;
			\item modelo está preparado para implementação da paralelização por plano;
		\end{itemize}
		\item experimento 2:
		\begin{itemize}
			\item capacidade do planejador se adaptar de acordo com a situção dos recursos;
			\item nós parcialmente ocupados: planos diferentes, mas com o mesmo número de passos;
		\end{itemize}
		\item experimento 3:
		\begin{itemize}
			\item criação de operadores mais elaborados: vários conjuntos de dados de entrada e saída;
		\end{itemize}
		\item geração de planos ótimos com variação do cenário;
		\item demora para propagação dos resultados;
	\end{itemize}
\end{frame}
