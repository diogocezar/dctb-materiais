\section[Introdução]{Introdução}

\begin{frame}
    \frametitle{Introdução}
    \begin{itemize}
		\item fluxos de trabalho científico: processamento e análise de dados de forma coordenada e integrada;
		\begin{itemize}
			\item encapsulamento de informações;
			\item processamento automático em um fluxo de execução;
			\item gerenciados por um sistema;
		\end{itemize}
		\item problema: carência de sistemas para execução de trabalhos científicos;
		\begin{itemize}
			\item escalonadores simplistas;
		\end{itemize}
		\item proposta: SWEP - \emph{Smart Workflow Execution by Planning};
		\begin{itemize}
			\item planejador (\emph{CRIKEY});
			\item ambiente P2P (\emph{PeerUnit});
			\item características:
			\begin{itemize}
				\item métricas flexíveis;
				\item técnicas de paralelização;
				\item proveniência e metadados;
			\end{itemize}
		\end{itemize}
	\end{itemize}
\end{frame}
