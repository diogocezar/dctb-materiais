\begin{frame}
    \frametitle{Trabalhos Relacionados}
	\begin{itemize}
		\item Ptolemy/Kepler:
		\begin{itemize}
			\item implementação da mesma operação sobre dados de diferentes formas (atores);
			\item modelo é capaz de selecionar qual a melhor implementação do operador para a execução em questão;
		\end{itemize}
		\item Pegasus:
		\begin{itemize}
			\item sistema é realimentado com metadados coletados em suas execuções;
		\end{itemize}
		\item SciCumulus:	
		\begin{itemize}
			\item inspirou a utilização de paralelismo: sobre dados e sobre execução;
		\end{itemize}
	\end{itemize}
\end{frame}
