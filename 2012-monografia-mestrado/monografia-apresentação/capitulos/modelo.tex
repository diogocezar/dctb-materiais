\begin{frame}
    \frametitle{Paralelismos}
    \begin{itemize}
		\item paralelismo sobre dados:
		\begin{itemize}
			\item divisão do processamento de um único fluxo de trabalho;
			\item desenvolvedor insere as configurações (número de nós ou tamanho dos blocos);
		\end{itemize}
		\item paralelismo sobre execução:
		\begin{itemize}
			\item paralelismo entre execução de múltiplos fluxos de trabalho;
			\item plano paralelo: pondera as dependências de dados;
			\item ordem correta de execução;
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Métrica Flexível}
    \begin{itemize}
		\item planejamento baseado nos custos das ações;
		\item oráculo:
		\begin{itemize}
			\item verifica-se a taxa de ocupação dos nós (livre, parcialmente ocupado ou ocupado);
			\item aplica-se custos pré-definidos para cada uma das situações;
		\end{itemize}
		\item utilização do planejamento de custos para aplicação de outras métricas:
		\begin{itemize}
			\item minimização do consumo de energia elétrica;
			\item maximização de aproveitamento dos recursos;
		\end{itemize}
		\item quantidade de trabalho pode variar de acordo com a situação dos recursos, em planos diferentes;
	\end{itemize}
\end{frame}
