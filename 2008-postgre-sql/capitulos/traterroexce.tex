\section{Tratamento De Erros e Exceções}

\begin{frame}
    \frametitle{Tratamento De Erros E Exceções}
    \begin{block}{Definição}
        Ações que causam anômalias nas aplicações.
    \end{block}
    \begin{itemize}
        \item Várias formas de tratamento;
        \item Plataforma .NET: Tratamento de excessões estruturadas;
        \item Objeto herdado de \emph{System.Exception} deve ser
        criado para representar as excessões;
        \item O tratamento pode acontecer com alguma excessão
        pré-definida.
    \end{itemize}
\end{frame}

\subsection{Bloco Try - Catch}

\begin{frame}[t,allowframebreaks]
    \frametitle{Bloco Try - Catch}
    \begin{block}{Definição}
        Uma ou mais instruções \emph{catch} são colocadas logo abaixo do
        bloco \emph{try} para interceptar uma excessão.
    \end{block}
    \begin{itemize}
        \item Dentro do bloco catch é encontrado o tratamento da
        excessão;
        \item Encontrado na forma hierárquica (cada \emph{catch} é
        verificado de acordo com a exceção);
        \item Um \emph{catch} pode estar isolado, tratando qualquer
        exceção.
    \end{itemize}
    \vspace{-2cm}

\end{frame}
