\begin{frame}
    \frametitle{A linguagem C\#}
    \begin{itemize}
        \item <1-> É uma linguagem da plataforma .NET;
        \item <2-> Derivada do C++;
        \item <3-> Orientada a objetos.
    \end{itemize}

\subsection{Características do C\#}

\begin{frame}
    \frametitle{Características do C\#}
    \begin{itemize}
        \item <1-> Simplicidade;
        \item <2-> Completamente orientada a objetos;
        \item <3-> Fortemente tipada;
        \item <4-> Tudo é um objeto;
        \item <5-> Controle de versões;
        \item <6-> Entre outros.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Características do C\#}
    Alguns detalhes que devem ser observados:
    \begin{itemize}
        \item <1-> Sensitive CASE, diferencia maiúsculas e minúsculas;
        \item <2-> As instruções terminam com ponto e virgula;
        \item <3-> As extensões dos arquivos são .cs;
    \end{itemize}
    \begin{block}<4->{Comentários}
        \small
        \textcolor[rgb]{0.00,0.50,0.00}{
        // Comenta apenas uma linha por vez \\
        \vspace{0.5cm}
        /* Este é um comentário de múltiplas linhas. \\
        Ele pode ser dividido Em muitas linhas */}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Tipos Valor e Tipos Referência}
    \begin{itemize}
            \item <1-> Tipos valor(\emph{value types}): dados em memória;
            \item <2-> Tipos referência(\emph{reference types}): referência para um valor;
            \item <3-> Tipos ponteiro(\emph{pointer types}): apontam para um endereço de memória.
        \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Conversão de tipos}
    \begin{itemize}
            \item <1-> Valores são convertidos sempre para o tipo de maior faixa de valores;
            \item <2-> A conversão dos tipos de ponto flutuante(float, double) para decimal causa erro;
            \item <3-> A conversão entre os tipos com sinal e sem sinal de valores inteiros com o mesmo tamanho causa erro.
    \end{itemize}
\end{frame}

            \item Para passar um argumento array para um método, especifique o nome do array sem usar colchetes.
            \item Para que um método receba um \emph{array}, a lista de parâmetros deve especificar que um array será recebido.
    \end{itemize}

    \texttt{\lstinputlisting[language=C, label=array5, caption={Passando arrays à métodos}]{cods/array5.txt}}

    \begin{block}{Propriedades e Métodos dos Arrays}
        \begin{itemize}
            \item \texttt{obj.Length} $\longrightarrow$ Tamanho do vetor;
            \item \texttt{Array.IndexOf(Array vetor, object value)} $\longrightarrow$ Procura a primeira ocorrência de valor em vetor;
            \item \texttt{Array.LastIndexOf(Array vetor, object value)} $\longrightarrow$ Procura a última ocorrência de valor em vetor;
            \item \texttt{Array.Sort(Array vetor)} $\longrightarrow$ Ordena um vetor crescentemente;
            \item \texttt{Array.Reverse(Array vetor)} $\longrightarrow$ Ordena um vetor decrescentemente.
        \end{itemize}
    \end{block}
